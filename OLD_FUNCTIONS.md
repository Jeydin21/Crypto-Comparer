```py
def getListDates(year, limit):
  datelist = []
  for num in range(1, limit + 1):
    month = 1
    for num in range(1, 13):
      date = datetime.date(year, num, 1)
      year = int(date.year)
      month = int(date.month)
      day = int(date.day)
      datelist.append(str(day) + '-' + str(month) + '-' + str(year))
      month += 1
    year += 1
  return datelist
```
